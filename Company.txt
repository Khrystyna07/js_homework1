const company = {
    name: 'Велика Компанія',
    type: 'Головна компанія',
    platform: 'Платформа для продажу квитків',
    sellsSolution: 'Рішення для продажу квитків',
    clients: [
        {
            name: 'Клієнт 1',
            type: 'subCompany',
            uses: 'ПО для продажу квитків',
            sells: 'Рішення для продажу квитків',
            partners: [
                {
                    name: 'Клієнт 1.1',
                    type: 'subSubCompany',
                    uses: 'Рішення для продажу квитків',
                    sells: 'Рішення для продажу квитків',
                },
                {
                    name: 'Клієнт 1.2',
                    type: 'subSubCompany',
                    uses: 'Рішення для продажу квитків',
                    sells: 'Рішення для продажу квитків',
                    partners: [
                        {
                            name: 'Клієнт 1.2.3',
                            type: 'subSubCompany',
                            uses: 'Рішення для продажу квитків',
                            sells: 'Рішення для продажу квитків',
                        }
                    ]
                }
            ]
        },
        {
            name: 'Клієнт 2',
            type: 'subCompany',
            uses: 'ПО для продажу квитків',
            sells: 'Рішення для продажу квитків'
        }
    ]
};

const companyNameInput = prompt("Введіть назву компанії:");

function findCompanyByName(name, company) {
    if (company.name === name) {
        return company;
    }
    if (company.clients) {
        for (let client of company.clients) {
            const found = findCompanyByName(name, client);
            if (found) return found;
        }
    }
    if (company.partners) {
        for (let partner of company.partners) {
            const found = findCompanyByName(name, partner);
            if (found) return found;
        }
    }
    return null;
}

const foundCompany = findCompanyByName(companyNameInput, company);
if (foundCompany) {
    console.log(`Компанія ${companyNameInput} знайдена:`, foundCompany);
} else {
    console.log(`Компанія ${companyNameInput} не знайдена.`);
}
